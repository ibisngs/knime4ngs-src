017    package org.apache.commons.lang3.text.translate;
018    
019    import java.io.IOException;
020    import java.io.Writer;
021    
022    /**
023     * Helper subclass to CharSequenceTranslator to allow for translations that 
024     * will replace up to one character at a time.
025     * 
026     * @since 3.0
027     * @version $Id: CodePointTranslator.java 1139924 2011-06-26 19:32:14Z mbenson $
028     */
029    public abstract class CodePointTranslator extends CharSequenceTranslator {
030    
031        /**
032         * Implementation of translate that maps onto the abstract translate(int, Writer) method. 
033         * {@inheritDoc}
034         */
035        @Override
036        public final int translate(CharSequence input, int index, Writer out) throws IOException {
037            int codepoint = Character.codePointAt(input, index);
038            boolean consumed = translate(codepoint, out);
039            if (consumed) {
040                return 1;
041            } else {
042                return 0;
043            }
044        }
045    
046        /**
047         * Translate the specified codepoint into another. 
048         * 
049         * @param codepoint int character input to translate
050         * @param out Writer to optionally push the translated output to
051         * @return boolean as to whether translation occurred or not
052         * @throws IOException if and only if the Writer produces an IOException
053         */
054        public abstract boolean translate(int codepoint, Writer out) throws IOException;
055    
056    }
